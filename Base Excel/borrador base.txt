create database cux;

use cux;

create table dependencias(
idDependencia int primary key, 
nombre varchar(100),
sector boolean,
correo varchar(50),
telefono bigint,
jefeInmediato varchar(100),
asesor varchar(100),
observaciones varchar(100));

create table alumno(
id int primary key,
fechaRegistro timestamp,
carrera varchar(10),
numCuenta int,
nombre varchar(100),
sexo varchar(10),
correoElectronico varchar(50),
telefono bigint,
ss varchar(1),
fechaInicio date,
fechaTermino date,
cartaPresentacion varchar(1),
1erInformeTri varchar(1),
2doInformeTri varchar(1),
3erInformeTri varchar(1),
InformeFinal varchar(1),
Terminacion varchar(1),
Evaluacion varchar(1),
SolicitudSS varchar(1),
foreign key (carrera) references carrera(nombre) on update cascade on delete cascade);

create table carrera(
nombre varchar(5) primary key,
creditos int);
